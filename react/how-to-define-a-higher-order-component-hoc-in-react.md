```jsx
// randomRGBColor.js
const randomRGBColor = () => {
    const makeColor = () => parseInt(Math.random() * 255, 10)
    const colorList = [,,,].fill().map(makeColor)
    const colorData = colorList.join(',')
    return colorData
}

export default randomRGBColor
```
